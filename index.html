function myFunction() { var SPREADSHEET_ID = "1TANL3h6Irb2hajRwP79t0vsPwv33yvhB47BAKGq7It8"; var POSSIBLE_SHEET_NAMES = ["Resultats", "Résultats", "Feuille 1", "Sheet1"]; var HEADERS = [ "ID soumission", "Date", "Prenom", "Nom", "Classe", "Note", "Total", "Pourcentage", "Detail reponses", "Reponses JSON", "Source" ]; function doPost(e) { return enregistrerResultat_(e); } function doGet(e) { if (e && e.parameter && e.parameter.payload_json) { return enregistrerResultat_(e); } return sortieJson_({ status: "ok", message: "Script operationnel. Utilisez un POST de formulaire pour enregistrer une note." }); } function enregistrerResultat_(e) { var lock = LockService.getScriptLock(); var locked = false; try { lock.waitLock(10000); locked = true; var data = lireDonnees_(e); if (!data || !data.prenom || !data.nom || !data.classe) { throw new Error("Donnees incompletes : prenom, nom et classe sont obligatoires."); } var sheet = obtenirFeuille_(); preparerEntetes_(sheet); var submissionId = data.submissionId || Utilities.getUuid(); if (dejaEnregistre_(sheet, submissionId)) { return sortieJson_({ status: "duplicate", message: "Resultat deja enregistre." }); } sheet.appendRow([ submissionId, data.date || new Date().toLocaleString("fr-FR"), data.prenom || "", data.nom || "", data.classe || "", Number(data.note || 0), Number(data.total || 10), Number(data.pourcentage || 0), data.details || "", data.answersJson || "", data.source || "ia-biais-qcm" ]); return sortieJson_({ status: "success", message: "Note enregistree.", id: submissionId }); } catch (err) { return sortieJson_({ status: "error", message: String(err) }); } finally { if (locked) lock.releaseLock(); } } function lireDonnees_(e) { var data = {}; if (e && e.parameter) { Object.keys(e.parameter).forEach(function (key) { data[key] = e.parameter[key]; }); if (e.parameter.payload_json) { try { var parsedFromForm = JSON.parse(e.parameter.payload_json); Object.keys(parsedFromForm).forEach(function (key) { data[key] = parsedFromForm[key]; }); } catch (ignore) {} } } if (Object.keys(data).length === 0 && e && e.postData && e.postData.contents) { try { data = JSON.parse(e.postData.contents); } catch (ignoreJson) { data = { raw: e.postData.contents }; } } return data; } function obtenirFeuille_() { var ss = SpreadsheetApp.openById(SPREADSHEET_ID); for (var i = 0; i < POSSIBLE_SHEET_NAMES.length; i++) { var found = ss.getSheetByName(POSSIBLE_SHEET_NAMES[i]); if (found) return found; } return ss.getSheets()[0] || ss.insertSheet("Resultats"); } function preparerEntetes_(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); } var firstCell = String(sheet.getRange(1, 1).getValue() || "").toLowerCase(); if (firstCell !== "id soumission") { sheet.insertColumnBefore(1); } sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]); sheet.getRange(1, 1, 1, HEADERS.length) .setFontWeight("bold") .setBackground("#1d4ed8") .setFontColor("#ffffff"); } function dejaEnregistre_(sheet, submissionId) { if (!submissionId || sheet.getLastRow() < 2) return false; var values = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues(); for (var i = 0; i < values.length; i++) { if (String(values[i][0]) === String(submissionId)) return true; } return false; } function sortieJson_(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } }
