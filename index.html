La Méthode des 3C

La clé d'or pour écrire des prompts efficaces et neutres

🎯

CLAIR

Formuler une demande précise, concise, sans fautes d'orthographe et avec des mots simples.

🏷️

CONTEXTUALISÉ

Donner un rôle à l'IA (ex: "Agis en historien"), lui décrire la situation et spécifier le format attendu.

⚖️

CORRECT / NEUTRE

Utiliser un langage objectif, sans termes subjectifs ou orientés, pour éviter d'influencer ou de biaiser l'IA.

// ================================================================
// SCRIPT CORRIGÉ — utilise doGet() avec paramètres URL
// Méthode GET : aucun problème de CORS ni de redirection POST
// ================================================================

function doGet(e) {
  // Si la requête contient des données élève, on enregistre la note
  if (e && e.parameter && e.parameter.nom) {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName("Résultats")
               || ss.getSheetByName("Resultats")
               || ss.getSheets()[0];

    // Créer les en-têtes si la feuille est vide
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["Date", "Prénom", "Nom", "Classe",
                       "Note", "Total", "Pourcentage", "Détails"]);
    }

    sheet.appendRow([
      e.parameter.date,
      e.parameter.prenom,
      e.parameter.nom,
      e.parameter.classe,
      e.parameter.note,
      e.parameter.total,
      e.parameter.pourcentage,
      e.parameter.details
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({ status: "success" }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // Requête sans données = test de connectivité
  return ContentService
    .createTextOutput("Script opérationnel ! Prêt à recevoir des notes.")
    .setMimeType(ContentService.MimeType.TEXT);
}
