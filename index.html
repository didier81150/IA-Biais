La Méthode des 3C

La clé d'or pour écrire des prompts efficaces et neutres

🎯

CLAIR

Formuler une demande précise, concise, sans fautes d'orthographe et avec des mots simples.

🏷️

CONTEXTUALISÉ

Donner un rôle à l'IA (ex: "Agis en historien"), lui décrire la situation et spécifier le format attendu.

⚖️

CORRECT / NEUTRE

Utiliser un langage objectif, sans termes subjectifs ou orientés, pour éviter d'influencer ou de biaiser l'IA.

// ============================================================
// COLLER CE CODE COMPLET DANS GOOGLE APPS SCRIPT
// Déployer comme Application Web :
//   - Exécuter en tant que : Moi
//   - Qui a accès       : Tout le monde
// ============================================================

function doPost(e) {
  // --- Autorisation CORS : indispensable pour les requêtes depuis le navigateur ---
  var output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);

  try {
    var ss = SpreadsheetApp.openById("1TANL3h6Irb2hajRwP79t0vsPwv33yvhB47BAKGq7It8");
    var sheet = ss.getSheetByName("Résultats")
              || ss.getSheetByName("Resultats")
              || ss.getSheetByName("Feuille 1")
              || ss.getSheetByName("Sheet1")
              || ss.getSheets()[0];

    // Créer les en-têtes si la feuille est vide
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(["Date", "Prénom", "Nom", "Classe", "Note /10", "Pourcentage", "Détail réponses"]);
      sheet.getRange(1, 1, 1, 7).setFontWeight("bold").setBackground("#4a90d9").setFontColor("#ffffff");
    }

    // Lire le corps de la requête (envoyé en text/plain pour éviter le preflight CORS)
    var raw = e.postData ? e.postData.contents : "";
    var data = JSON.parse(raw);

    // Ajouter la ligne de résultat
    sheet.appendRow([
      data.date        || new Date().toLocaleString("fr-FR"),
      data.prenom      || "",
      data.nom         || "",
      data.classe      || "",
      data.note + " / " + data.total,
      data.pourcentage + " %",
      data.details     || ""
    ]);

    output.setContent(JSON.stringify({ status: "success", message: "Note enregistrée." }));
  } catch (err) {
    output.setContent(JSON.stringify({ status: "error", message: err.toString() }));
  }

  return output;
}

// doGet permet de tester que le script est actif depuis le navigateur
function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ status: "ok", message: "Script opérationnel !" }))
    .setMimeType(ContentService.MimeType.JSON);
}
